function u_prime(c; γ=2.0)
    return c^(-γ)
end


function at(p, c)
    @unpack R, y, T, a = p
    for t in 1:T
        a[t+1] = R * a[t] + y[t] - c[t]
    end    
end

# Objective function
function objective!(p, F, x)
    @unpack β, R, y, a, T = p
    #c = x
    # F = similar(x)
    for t in 1:T-1
        F[t] = u_prime(x[t]) - β * R * u_prime(x[t+1])
        a[t+1] = R * a[t] + y[t] - x[t]
    end
    a[end] = R * a[end-1] + y[end] - x[end]
    F[end] = a[end]
    return F
end

function objective(p, x)
    objective!(p, similar(x), x)
end
